def acf2(ts):
    plt.figure()
    plt.subplot(211)
    plot_acf(ts, ax=plt.gca(), lags = 30)
    plt.subplots_adjust(hspace=0.5)
    plt.subplot(212)
    plot_pacf(ts, ax=plt.gca(), lags = 30, method = 'ywm')
    plt.show()
